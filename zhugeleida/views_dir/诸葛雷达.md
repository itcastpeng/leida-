程序展示: 生成 token 和 pwd 、timestamp 
def str_encrypt(pwd):
    """
    :param pwd: 密码
    :return:
    """
    pwd = str(pwd)
    hash = hashlib.md5()
    hash.update(pwd.encode())
    return hash.hexdigest()
def get_token(pwd):
    tmp_str = str(int(time.time()*1000)) + pwd  [加密后的密码]
    token = str_encrypt(tmp_str)
    return token
